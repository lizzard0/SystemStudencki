@Entity
@Table(name = "platnosc")
public class Platnosc {

    @Id
    @GeneratedValue
    @Column(name = "id")
    private Long id;

    public Platnosc() {
    }

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }
}
