@Entity
@Table(name = "ocena_czastkowa")
public class OcenaCzastkowa {

    @Id
    @GeneratedValue
    @Column(name = "id")
    private Long id;

    public OcenaCzastkowa() {
    }

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }
}
